resource "aws_instance" "web" {
  ami                         = var.ami_id
  instance_type               = "t2.micro"
  subnet_id                   = var.subnet_id
  security_groups             = [var.security_group_id]

  root_block_device {
    volume_size               = var.root_volume_size
  }

  ebs_block_device {
    device_name               = "/dev/sdh"
    volume_size               = var.log_volume_size
    delete_on_termination     = true
  }

  tags = {
    Name = "web-instance"
  }

  provisioner "remote-exec" {
    inline = [
      "sudo apt-get update",
      "sudo apt-get install -y apache2", # Install Apache web server (you can change this to your desired web server)
      "sudo mkdir /var/log/custom_logs", # Creating a directory for custom logs
      # Add any additional configuration or setup steps here as needed
    ]
  }
}
